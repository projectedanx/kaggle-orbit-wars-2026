```python
import math
import numpy as np
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class MCTSNode:
    """
    Honest implementation of kernel-smoothed MCTS for continuous action spaces.
    Based on actual literature: MCTS-SR, RAVE, and kernel UCB.
    """
    action: float           # continuous action value (e.g., aim angle)
    visits: int = 0
    total_reward: float = 0.0
    children: list = field(default_factory=list)
    parent: Optional['MCTSNode'] = None

    def mean_reward(self) -> float:
        if self.visits == 0:
            return 0.0
        return self.total_reward / self.visits


def kernel_similarity(a1: float, a2: float, gamma: float = 0.5) -> float:
    """
    RBF kernel: measures similarity between two continuous actions.
    This genuinely helps generalize across nearby actions in continuous spaces.
    gamma controls bandwidth - real hyperparameter with real effects.
    """
    return math.exp(-gamma * (a1 - a2) ** 2)


def kernel_smoothed_value(
    query_action: float,
    action_history: list[tuple[float, float, int]],  # (action, reward, visits)
    gamma: float = 0.5,
    min_denominator: float = 1e-9
) -> float:
    """
    Kernel regression estimate of action value.
    Weighted average of nearby actions' rewards.
    
    This is legitimate: it reduces sample complexity in continuous action spaces
    by sharing information across similar actions.
    """
    weighted_reward = 0.0
    weight_sum = 0.0

    for action, reward, visits in action_history:
        weight = kernel_similarity(query_action, action, gamma) * visits
        weighted_reward += weight * reward
        weight_sum += weight

    if weight_sum < min_denominator:
        return 0.0

    return weighted_reward / weight_sum


def ucb1_tuned(
    node: MCTSNode,
    parent_visits: int,
    exploration_constant: float = math.sqrt(2)
) -> float:
    """
    UCB1-Tuned: actual algorithm from Auer et al. 2002.
    Not metaphysically entangled with HTTP status codes.
    """
    if node.visits == 0:
        return float('inf')

    exploitation = node.mean_reward()
    exploration = exploration_constant * math.sqrt(
        math.log(parent_visits) / node.visits
    )
    return exploitation + exploration


def kernel_ucb(
    query_action: float,
    node: MCTSNode,
    action_history: list[tuple[float, float, int]],
    parent_visits: int,
    alpha: float = 0.5,
    exploration_constant: float = math.sqrt(2)
) -> float:
    """
    Blend of UCB1 (exploitation/exploration) and kernel estimate.
    alpha controls interpolation between direct node value and kernel estimate.
    
    This is the legitimate kernel-smoothed MCTS idea:
    - Direct node value: unbiased but high variance
    - Kernel estimate: biased but lower variance
    alpha trades these off.
    """
    direct_ucb = ucb1_tuned(node, parent_visits, exploration_constant)
    kernel_estimate = kernel_smoothed_value(query_action, action_history)

    return alpha * direct_ucb + (1 - alpha) * kernel_estimate
```
