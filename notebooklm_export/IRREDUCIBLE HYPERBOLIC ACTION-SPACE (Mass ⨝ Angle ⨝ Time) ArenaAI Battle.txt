        target.y + target.orbital_radius * math.sin(target.theta_0 + angular_w * t)

    return math.sqrt((a[0]-b[0])**2 + (a[1]-b[1])**2)

def solve_intercept_time(source: Planet, target: Planet, v: float, angular_w: float,
                         t0: float = 0.0, max_iter: int = 50, tol: float = 1e-6) -> Optional[float]:
    """
    Solve ||P_t(t) - P_s|| - v*t = 0 by Newton–Raphson.
    Returns t_collapse or None if no root found in a reasonable interval.
    """
    s = np.array([source.x, source.y])
    t = t0
    for _ in range(max_iter):
        pt = target_position(target, t, angular_w)
        f = distance(pt, s) - v * t

        # v_t = angular_w * target.orbital_radius * np.array([-sin, cos])

        dfdt = ( (pt - s) @ v_t ) / distance(pt, s) - v
        if abs(dfdt) < 1e-12:
            return None
        t_new = t - f / dfdt
        if abs(t_new - t) < tol and abs(f) < tol:
            return t_new
        t = t_new
        if t < 0:
            t = 0.0
    return None

        p = s + v * t * dir_vec
        d = distance(p, np.array([sun.x, sun.y]))
        d_min = min(d_min, d)
    return d_min

    for i in range(3, root + 1, 2):

    """v(m) = 1 + 5·(ln(m)/ln(1000))^1.5 for m ≥ 2; v(1) = 1.0.
    
    Monotonically increasing, strictly concave.
    This single function is what entangles mass with geometry:
    every change in m deforms the entire intercept surface.
    """
    if m <= 1:
        return 1.0
    return 1.0 + 5.0 * (math.log(m) / math.log(1000)) ** 1.5

# ─────────────────────────────────────────────────────────
# TRANSCENDENTAL INTERCEPT SOLVER
# ─────────────────────────────────────────────────────────
def _intercept_residual(t: float, sx: float, sy: float,
                        r: float, theta0: float, omega: float,
                        v: float) -> float:
    """f(t) = ||P_target(t) - P_source||² - (v·t)²
    
    Roots of f are intercept times. The transcendental trig terms
    make this impossible to solve algebraically — the coupling
    between mass (via v) and time is analytically irreducible.
    """
    tx = SUN_X + r * math.cos(theta0 + omega * t) - sx
    ty = SUN_Y + r * math.sin(theta0 + omega * t) - sy
    return tx * tx + ty * ty - (v * t) ** 2


def solve_intercept_time(source_pos: Tuple[float, float],
                         target: Planet, v: float, omega: float,
                         t_max: float = 500.0,
                         n_scan: int = 2000) -> Optional[float]:
    """Find the earliest positive t where fleet arrives at target.
    
    Uses sign-change scanning + Brent's method on the residual.
    Returns None if no intercept exists within t_max.
    """
    sx, sy = source_pos
    r, th0 = target.orbital_radius, target.theta_0
    f = lambda t: _intercept_residual(t, sx, sy, r, th0, omega, v)

    ts = np.linspace(1e-4, t_max, n_scan)
    fs = np.array([f(t) for t in ts])

    for i in range(len(fs) - 1):
        if fs[i] * fs[i + 1] < 0:
            try:
                root = brentq(f, ts[i], ts[i + 1], xtol=1e-12)
                if root > 1e-4:
                    return root
            except ValueError:
                continue
    return None

# ─────────────────────────────────────────────────────────
# SUN OCCLUSION GEOMETRY
# ─────────────────────────────────────────────────────────
def min_distance_to_sun(sx, sy, tx, ty) -> float:
    """Closest approach of the line segment (sx,sy)→(tx,ty) to the Sun."""
    abx, aby = tx - sx, ty - sy
    apx, apy = SUN_X - sx, SUN_Y - sy
    ab2 = abx * abx + aby * aby
    if ab2 < 1e-12:
        return math.hypot(apx, apy)
    t = max(0.0, min(1.0, (apx * abx + apy * aby) / ab2))
    return math.hypot(SUN_X - (sx + t * abx), SUN_Y - (sy + t * aby))


def path_is_blocked(sx, sy, tx, ty, margin: float = 0.5) -> bool:
    return min_distance_to_sun(sx, sy, tx, ty) < SUN_R + margin

# ─────────────────────────────────────────────────────────
# INTERCEPT SOLVER WITH OCCLUSION FILTERING
# ─────────────────────────────────────────────────────────
def find_clear_intercept(source_pos: Tuple[float, float],
                         target: Planet, v: float, omega: float,
                         t_max: float = 500.0,
                         n_scan: int = 4000) -> Optional[Tuple[float, float, float]]:
    """Find the earliest intercept whose straight-line path clears the Sun.
    
    When the direct intercept is occluded, we do NOT guess a tangent —
    we search for a LATER root of the intercept equation where the
    target has orbited into a non-occluded position. This preserves
    the coupling: the alternative intercept still depends on v(m).
    
    Returns (t_intercept, target_x, target_y) or None.
    """
    sx, sy = source_pos
    r, th0 = target.orbital_radius, target.theta_0
    f = lambda t: _intercept_residual(t, sx, sy, r, th0, omega, v)

    ts = np.linspace(1e-4, t_max, n_scan)
    fs = np.array([f(t) for t in ts])

    for i in range(len(fs) - 1):
        if fs[i] * fs[i + 1] < 0:
            try:
                root = brentq(f, ts[i], ts[i + 1], xtol=1e-12)
            except ValueError:
                continue
            if root < 1e-4:
                continue
            # Compute intercept position and check occlusion
            tgt = target.position_at(root, omega)
            if not path_is_blocked(sx, sy, tgt[0], tgt[1]):
                return (root, tgt[0], tgt[1])
            # Otherwise: this root is occluded, continue to next root
    return None

# ─────────────────────────────────────────────────────────
# THE FULL ACTION COMPUTATION
# ─────────────────────────────────────────────────────────
def compute_action(source: Planet, target: Planet,
                   num_ships: int, omega: float,
                   t_now: float = 0.0) -> Optional[dict]:
    """
    The irreducible action: (angle, ships) → intercept.
    
    You cannot precompute a table of "best angles per target" 
    independently of ship count. The table itself deforms with m.
    """
    if num_ships < 1:

    theta = math.atan2(ty - src[1], tx - src[0])

        "from": source.id,
        "to": target.id,
        "angle": theta,
        "ships": num_ships,
        "velocity": v,
        "travel_time": t_int,
        "intercept_pos": (tx, ty),
    }
